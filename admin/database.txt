CREATE DATABASE joinweb DEFAULT CHARACTER SET utf8 COLLATE utf8_unicode_ci;

USE joinweb;

CREATE TABLE users
(
	id_user int PRIMARY KEY AUTO_INCREMENT,
    username varchar(30),
    password varchar(30)
);

CREATE TABLE article
(
	id_article int PRIMARY KEY AUTO_INCREMENT,
    title varchar(31),
    subtitle varchar(63),
    article varchar(4095),
    img varchar(255)
);
